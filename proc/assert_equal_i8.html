  subroutine assert_equal_i8(this, i1, i2)
    class(tester_t), intent(inout)        :: this !< The tester.
    integer(int8),   intent(in)           :: i1   !< Value to compare.

    this% n_tests = this% n_tests + 1
    if (i1 .ne. i2) then
       this% n_errors = this% n_errors + 1
    end if
